	Tag "Health Bonus"
	Inventory.PickupMessage "$GOTHTHBONUS"

			TNT1 A 0 A_CheckSight ("Spawn3")
			TNT1 A 0 A_Jump(128,5)
			TNT1 A 0 A_Jump(128,5)
			TNT1 A 0 A_Jump(128,5)
			TNT1 A 0 A_Jump(128,5)
			HSPH ABCDEFGH 2 BRIGHT A_SpawnItem("BlueFlareSmallItem",0,7)
			Loop
		Spawn3:
			HSPH A 9 BRIGHT
			Goto Spawn2
		Vanilla:
			HSPH ABCDEFGH 2 BRIGHT
			Loop
	}
}
